#include <iostream>
#include <vector>
#include <queue>

using namespace std;

vector<int> kahnTopologicalSort(int V, vector<int> adj[]) {
    vector<int> inDegree(V, 0);
    vector<int> topoOrder;

    // 1. Calculate in-degrees for all vertices
    for (int i = 0; i < V; i++) {
        for (int neighbor : adj[i]) {
            inDegree[neighbor]++;
        }
    }

    // 2. Push all vertices with in-degree 0 into the queue
    queue<int> q;
    for (int i = 0; i < V; i++) {
        if (inDegree[i] == 0) {
            q.push(i);
        }
    }

    // 3. Process the queue
    while (!q.empty()) {
        int u = q.front();
        q.pop();
        topoOrder.push_back(u);

        // For every neighbor, reduce its in-degree
        for (int v : adj[u]) {
            inDegree[v]--;
            
            // If in-degree becomes 0, add to queue
            if (inDegree[v] == 0) {
                q.push(v);
            }
        }
    }

    // 4. Cycle Check
    if (topoOrder.size() != V) {
        return {}; // Return empty if a cycle exists
    }

    return topoOrder;
}
