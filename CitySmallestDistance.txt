#include <bits/stdc++.h>
using namespace std;

int findTheCity(int n, vector<vector<int>>& edges, int distanceThreshold) {
    const int INF = 1e9;
    
    vector<vector<int>> dist(n, vector<int>(n, INF));

    // initialize
    for (int i = 0; i < n; i++) {
        dist[i][i] = 0;
    }

    for (auto &e : edges) {
        int u = e[0], v = e[1], w = e[2];
        dist[u][v] = w;
        dist[v][u] = w;
    }

    // Floyd-Warshall
    for (int k = 0; k < n; k++) {
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (dist[i][k] + dist[k][j] < dist[i][j]) {
                    dist[i][j] = dist[i][k] + dist[k][j];
                }
            }
        }
    }

    int minCount = INT_MAX;
    int city = -1;

    for (int i = 0; i < n; i++) {
        int count = 0;

        for (int j = 0; j < n; j++) {
            if (i != j && dist[i][j] <= distanceThreshold) {
                count++;
            }
        }

        // pick smallest count, if tie pick larger index
        if (count <= minCount) {
            minCount = count;
            city = i;
        }
    }

    return city;
}

int main() {
    int n = 4;
    vector<vector<int>> edges = {
        {0,1,3},
        {1,2,1},
        {2,3,4},
        {0,3,7}
    };
    int threshold = 4;

    cout << findTheCity(n, edges, threshold);
    return 0;
}
